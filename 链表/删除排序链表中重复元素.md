var deleteDuplicates = function(head) {
  if (!head) return head
  const dummy = new ListNode(0, head)
  while (head.next) {
    // 如果当前头部节点val 等于 下一个节点val
    // 则指针指向下一个的下一个节点，相当于删除了head.next节点
    // 否则就将指针移动到下一个节点
    if (head.val === head.next.val) {
      head.next = head.next.next
    } else {
      head = head.next
    }
  }
  return dummy.next
}
